# Тестовый скрипт для Этапа 1 - REPL
# Демонстрация работы всех команд

# Команда ls без аргументов
ls

# Команда ls с аргументами
ls -l
ls -a

# Команда cd без аргументов (домой)
cd

# Команда cd с относительным путем
cd scripts

# Команда cd с абсолютным путем
cd /home/user

# Команда cd с переходом на уровень выше
cd ..
cd ../..

# Команда cd с переходом в корень
cd /

# Попытка несуществующей команды
invalid_command

# Команда exit с аргументами (должна показать предупреждение)
exit param1 param2
